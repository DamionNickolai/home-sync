CHANGE BRANCH:
git checkout dev

SWISS ARMY KNIFE TO MAKE CHANGES:
git switch dev

CLOSE APP:
Ctrl + c

QUICK SAVE:
Ctrl + s

CREATE VIRTUAL ENV:
python -m venv venv

CREATE LOCAL .ENV FROM TEMPLATE:
Copy-Item maintenance/.env.example .env

BACKFILL AUTH IDS:
Run locally after setting SUPABASE_URL and SUPABASE_SERVICE_KEY:
python maintenance/backfill_auth_user_ids.py
python maintenance/backfill_auth_user_ids.py --apply

VALIDATE LOCAL MATCHES CLOUD:
git fetch origin dev
git status
VALIDATE ALL CLOUD:
git fetch --all
git branch -vv
STORE AND MERGE LOCAL UPDATES (Do this if you have made changes locally but origin is ahead):
git stash
git fetch origin dev
git status
git stash pop
JOINT WORKSPACE (HOME + GET FIT):
code home-sync-get-fit.code-workspace

SYNC BOTH APPS FAST:
git -C . fetch origin dev
git -C . status
git -C ../get-fit-together fetch origin dev
git -C ../get-fit-together status

PUSH A SMALL FIX TO DEV STAGING:
git add auth.py
git commit --amend --no-edit
git push --force origin dev

AUTO PUSH FROM LOCAL TO CLOUD:
python push_to_dev.py
python push_to_prod.py

INSTALL PACKAGES:
pip install -r requirements.txt

RUN DATABASE MIGRATIONS:
python maintenance/migrate_projects.py

BACKFILL ENCRYPTED SESSION TOKENS:
.\venv\Scripts\python.exe maintenance/migrate_user_sessions_refresh_tokens.py
.\venv\Scripts\python.exe maintenance/migrate_user_sessions_refresh_tokens.py --apply

ENCRYPTION KEY NOTE:
Home Sync and Get Fit Together must use the same ENCRYPTION_KEY in production because they share the user_sessions table.

ENTER VIRTUAL ENV:
.\venv\Scripts\activate

RUN APP:
python -m streamlit run home_sync.py
