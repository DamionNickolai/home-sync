CHANGE BRANCH:
git checkout dev

SWISS ARMY KNIFE TO MAKE CHANGES:
git switch dev

CLOSE APP:
Ctrl + c

QUICK SAVE:
Ctrl + s

CREATE VIRTUAL ENV:
python -m venv venv

VALIDATE LOCAL MATCHES CLOUD:
git fetch origin dev
git status

AUTO PUSH FROM LOCAL TO CLOUD:
python push_to_dev.py
python push_to_prod.py

PUSH A SMALL FIX TO DEV STAGING:
git add auth.py
git commit --amend --no-edit
git push --force origin dev

INSTALL PACKAGES:
pip install -r requirements.txt

ENTER VIRTUAL ENV:
.\venv\Scripts\activate

RUN APP:
python -m streamlit run home_sync.py
